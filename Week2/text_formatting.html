       def get_user_data(user_uid, userRole):
    """ This is a function that get the user data """

    if not user_uid and not userRole:
        return None

    redis_key = f'Fetched_data/{userRole}/{user_uid}'
    current_app.logger.info(f"Querying Redis with key: {redis_key}")

    try:
        # Check if key exists
        if not current_app.redis_client.exists(redis_key):
            current_app.logger.error(f"Key does not exist in Redis: {redis_key}")
            return None

        # Check cache for user data
        user_data = current_app.redis_client.get(redis_key)

            current_app.logger.info(f"Raw UserData from Redis: {user_data} (Type: {type(user_data)})")
